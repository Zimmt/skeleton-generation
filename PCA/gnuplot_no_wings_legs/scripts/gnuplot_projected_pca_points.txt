# 2D plots of projection of PCA data on first three eigenvectors

set terminal pdf

set xrange [-2:2]
set yrange [-2:2]
set zrange [-2:2]

set xlabel "first eigenvector"
set ylabel "second eigenvector"
set output "projection_eigenvectors12.pdf"
plot '../input/projected_pcaPoints.txt' using 1:2

set ylabel "third eigenvector"
set output "projection_eigenvectors13.pdf"
plot '../input/projected_pcaPoints.txt' using 1:3

set xlabel "second eigenvector"
set output "projection_eigenvectors23.pdf"
plot '../input/projected_pcaPoints.txt' using 2:3
