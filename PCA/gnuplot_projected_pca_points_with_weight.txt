# 2D plots of projection of PCA data on first three eigenvectors

set terminal pdf

set xrange [-0.5:0.5]
set yrange [-2:2]

set xlabel "weight"
set ylabel "first eigenvector"
set output "projection_weight_and_eigenvector1.pdf"
plot 'projected_pcaPoints_with_weight.txt' using 3:1

set ylabel "second eigenvector"
set output "projection_weight_and_eigenvector2.pdf"
plot 'projected_pcaPoints_with_weight.txt' using 3:2
